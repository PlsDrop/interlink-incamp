        public int x;
        public int y;

        {
            Console.WriteLine(this);
        }

        public override string ToString()
        {
            return $"(x: {x}, y: {y})";
        }

        public Point Move(int offset)
        {
            return new Point() {
                x = this.x + offset,
                y = this.y + offset,
            };
        }
    }

    class Rect
    {
        public Point topLeft;
        public int width;
        public int height; 

        public void Print()
        {
            Console.WriteLine($"[{topLeft}; {width} x {height}]");
        }

        public void Scale(int factor)
        {
            width *= factor;
            height *= factor;
        }

    }
}
